  // 모든 문장을 테마별로 반환
  static List<String> getSentencesByTheme(SentenceTheme theme) {
    switch (theme) {
      case SentenceTheme.healing:
        return healingSentences;
      case SentenceTheme.motivation:
        return motivationSentences;
      case SentenceTheme.love:
        return loveSentences;
      case SentenceTheme.literature:
        return literatureSentences;
      case SentenceTheme.daily:
        return dailySentences;
      case SentenceTheme.friendship:
        return friendshipSentences;
      case SentenceTheme.movie:
        return movieSentences;
      case SentenceTheme.kpop:
        return kpopSentences;
      case SentenceTheme.todayMessage:
        return todayMessageSentences;
      case SentenceTheme.basic:
        return basicPracticeSentences.expand((list) => list).toList();
      case SentenceTheme.all:
      default:
        return [
          ...healingSentences,
          ...motivationSentences,
          ...loveSentences,
          ...literatureSentences,
          ...dailySentences,
          ...friendshipSentences,
          ...movieSentences,
          ...kpopSentences,
          ...todayMessageSentences,
        ];
    }
  }

  // 레벨별 기본 연습 문장 반환 (레벨 1-20)
  static List<String> getPracticeTextsByLevel(int level) {
    if (level < 1 || level > 20) return ['연습 문장'];
    
    int arrayIndex = level - 1;
    if (arrayIndex < basicPracticeSentences.length) {
      return List.from(basicPracticeSentences[arrayIndex]);
    }
    return ['고급 연습 문장'];
  }

  // 레벨별 테마 문장 반환 (테마 + 레벨 조합)
  static List<String> getPracticeTextsByThemeAndLevel(SentenceTheme theme, int level) {
    if (theme == SentenceTheme.basic) {
      return getPracticeTextsByLevel(level);
    }
    
    List<String> themeSentences = getSentencesByTheme(theme);
    
    // 레벨에 따라 문장 길이 필터링
    List<String> filteredSentences = [];
    
    for (String sentence in themeSentences) {
      int charCount = sentence.length;
      
      if (level <= 5 && charCount <= 15) {
        // 초급: 15자 이하
        filteredSentences.add(sentence);
      } else if (level <= 10 && charCount <= 30) {
        // 중급: 30자 이하  
        filteredSentences.add(sentence);
      } else if (level <= 15 && charCount <= 50) {
        // 고급: 50자 이하
        filteredSentences.add(sentence);
      } else if (level > 15) {
        // 최고급: 모든 길이
        filteredSentences.add(sentence);
      }
    }
    
    // 필터링 결과가 없으면 모든 문장 반환
    return filteredSentences.isEmpty ? themeSentences : filteredSentences;
  }

  // 감정별 문장 반환
  static List<String> getSentencesByEmotion(EmotionType emotion) {
    switch (emotion) {
      case EmotionType.happy:
        return [...dailySentences, ...friendshipSentences];
      case EmotionType.sad:
        return [...healingSentences];
      case EmotionType.tired:
        return [...healingSentences];
      case EmotionType.motivated:
        return [...motivationSentences];
      case EmotionType.grateful:
        return [...friendshipSentences, ...loveSentences];
      case EmotionType.lonely:
        return [...healingSentences, ...friendshipSentences];
      case EmotionType.excited:
        return [...loveSentences, ...motivationSentences];
      case EmotionType.all:
      default:
        return getSentencesByTheme(SentenceTheme.all);
    }
  }

  // 오늘의 문장 (날짜 기반)
  static String getTodaySentence() {
    final today = DateTime.now();
    final dayOfYear = today.difference(DateTime(today.year, 1, 1)).inDays;
    final allSentences = getSentencesByTheme(SentenceTheme.all);
    return allSentences[dayOfYear % allSentences.length];
  }
